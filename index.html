🚀 Service Locator + DI + Singleton

1️⃣ SINGLETON - ServiceLocator


// ← SINGLETON: Una sola instancia global
class ServiceLocator {
  static instance = null;                    // ← Memoria estática
  services = new Map();                      // ← Registro de servicios

  constructor() {
    if (ServiceLocator.instance) {           // ← ¿Ya existe?
      return ServiceLocator.instance;        // ← RETORNA la existente
    }
    ServiceLocator.instance = this;          // ← Crea SOLO UNA vez
  }

  static getInstance() {                     // ← Acceso global
    return new ServiceLocator();
  }

  register(name, service) {                  // ← REGISTRAR servicio
    this.services.set(name, service);
  }

  get(name) {                                // ← OBTENER servicio (DI)
    const service = this.services.get(name);
    if (!service) throw new Error(`Servicio ${name} no encontrado`);
    return service;
  }
}

2️⃣ SERVICIOS con Inyección de Dependencias


// ← SERVICIO 1: Logger
class LoggerService {
  log(message) {
    const msg = `[LOG] ${new Date().toISOString()}: ${message}`;
    console.log(msg);
    return msg;                              // ← Retorna para mostrar
  }
  
  error(message) {
    const msg = `[ERROR] ${message}`;
    console.error(msg);
    return msg;
  }
}

// ← SERVICIO 2: UserService (DEPENDE de Logger)
class UserService {
  constructor() {
    this.logger = ServiceLocator.getInstance().get('logger');  // ← 🔥 DI AUTOMÁTICA
  }
  
  createUser(name, email) {
    this.logger.log(`Creando: ${name}`);                      // ← Usa Logger inyectado
    const user = {
      id: Math.floor(Math.random() * 1000),
      name, email
    };
    this.logger.log(`Usuario creado: ${user.id}`);
    return user;
  }
  
  getUser(id) {
    this.logger.log(`Buscando usuario ${id}`);
    return { id, name: `Usuario ${id}`, email: `user${id}@app.com` };
  }
}

// ← SERVICIO 3: EmailService (DEPENDE de Logger)
class EmailService {
  constructor() {
    this.logger = ServiceLocator.getInstance().get('logger');  // ← 🔥 DI AUTOMÁTICA
  }
  
  sendWelcome(user) {
    this.logger.log(`📧 Enviando email a ${user.email}`);
    return `✅ Email enviado a ${user.email}`;
  }
}

3️⃣ USO COMPLETO - Inicialización


// ← Inicializar TODO
function initializeApp() {
  const locator = ServiceLocator.getInstance();     // ← SINGLETON
  
  locator.register('logger', new LoggerService());  // ← REGISTRAR Logger
  locator.register('userService', new UserService()); // ← UserService (tiene Logger)
  locator.register('emailService', new EmailService()); // ← EmailService (tiene Logger)
  
  return locator;
}
